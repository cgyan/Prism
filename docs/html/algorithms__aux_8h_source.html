730 void
731 p_heapify_aux(RandomAccessIterator node, RandomAccessIterator first, RandomAccessIterator last) {
732 
733  int nodeIndex = node-first;
734  int leftIndex = 2 * nodeIndex + 1;
735  int rightIndex = 2 * nodeIndex + 2;
736 
737  RandomAccessIterator leftNode = first + leftIndex;
738  RandomAccessIterator rightNode = first + rightIndex;
739  RandomAccessIterator largestNode = node;
740 
741  if (leftNode < last && *largestNode < *leftNode)
742  largestNode = leftNode;
743 
744  if (rightNode < last && *largestNode < *rightNode)
745  largestNode = rightNode;
746 
747  if (node != largestNode) {
748  swap(*node, *largestNode);
749  p_heapify(largestNode, first, last);
750  }
