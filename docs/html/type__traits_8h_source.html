105 // ConditionalType
106 //============================================================================================
107 // conditional_type is a struct that has a single member called type.
108 // If condition is true then type's data type is set to Type1's data type.
109 // If condition is false then type's data type is set to Type2's data type.
110 //
111 // prism::conditional_type<true, int, String>::type t; // type of t is int
112 // prism::conditional_type<false, int, String>::type t; // type of t is String
113 template <bool condition, typename T1, typename T2>
114 struct ConditionalType {
115  typedef T1 type;
116 };
117 
118 template <typename T1, typename T2>
119 struct ConditionalType<false, T1, T2> {
120  typedef T2 type;
121 };
122 
123 template <bool condition, typename T1, typename T2>
124 using ConditionalType_t = typename prism::ConditionalType<condition,T1,T2>::type;
