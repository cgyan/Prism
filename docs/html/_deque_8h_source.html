464  fillInsert(iterator pos, const int numElements, const T& value) {
465  int insertIndex = pos-begin;
466 
467  // insert at beginning
468  if (pos.current == begin.current) {

472  }
473  // inserting at end

475  int oldSize = size();
476  reserveElementsAtBack(numElements);
477  prism::uninitialized_fill(begin+oldSize, begin+oldSize+numElements, value);
