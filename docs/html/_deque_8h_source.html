74  Self&
75  operator+=(const int n)
76  {
77  int elementIndex = n + (current-start);
78  if (elementIndex >= 0 && elementIndex < prism_deque_bucket_size)
79  current += n;
80  else {
81  int bucketOffset = 0;
82  if (elementIndex > 0) bucketOffset = elementIndex / prism_deque_bucket_size;
83  else bucketOffset = -((-elementIndex-1) / prism_deque_bucket_size) - 1;
84 
85  buckets += bucketOffset;
86  start = *buckets;
87  end = start + prism_deque_bucket_size;
88  current = start + (elementIndex - bucketOffset * prism_deque_bucket_size);
89  }
90  return *this;
91  }

368 template <class T>
369 void
370 DequeData<T>::
371 initializeStorage(const int numElements) {
372  int numBuckets = numElements / prism_deque_bucket_size + 1;
373 
374  storage.start = storage.allocateStorage(numBuckets);
375  storage.finish = storage.start + numBuckets;
376 
377  storage.createBuckets(storage.start, storage.finish);
378 
379  begin.buckets = storage.start;
380  begin.start = *storage.start;
381  begin.current = begin.start
382  + (numBuckets * prism_deque_bucket_size) / 2
383  - numElements / 2;
384  begin.end = begin.start + prism_deque_bucket_size;
385  end = begin + numElements;
386 }
