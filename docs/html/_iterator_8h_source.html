178 /****************************************************************************************************************
179  * Special iterator for the List container (implemented as a bidirectional iterator. A list contains a linked
180  * list of nodes and each node contains the actual values stored by the user. As such the iterator needs to return that
181  * value and not the node as that is part of the private implementation.
182  ****************************************************************************************************************/
183 template <class Node, class T>
184 struct ListIterator {
185  typedef T value_type;
186  typedef T& reference;
187  typedef T* pointer;
188  typedef bidirectional_iterator_tag iterator_category;
189  typedef std::ptrdiff_t difference_type;
190 
191  Node * node;
192  ListIterator() : node(0) {}
193  ListIterator(Node * node) : node(node) {}
194  ListIterator(const ListIterator & copy) { node = copy.node; }
195  reference operator*() { return node->value; }
196  pointer operator->() { return &node->value; }
197  ListIterator & operator++() { node = node->next; return *this; }
198  ListIterator operator++(int junk) { Node * n = node; node = node->next; return n; }
199  ListIterator & operator--() { node = node->prev; return *this; }
200  ListIterator operator--(int junk) { Node * n = node; node = node->prev; return n; }
201  ListIterator & operator=(const ListIterator & rhs) { node = rhs.node; return *this;}
202  bool operator!=(const ListIterator & rhs) { return node != rhs.node; }
203  bool operator==(const ListIterator & rhs) const { return node == rhs.node; }
204
