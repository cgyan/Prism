26 /****************************************************************************************************************
27  * Special iterator for the List container (implemented as a bidirectional iterator.
28  * A list contains a linked list of nodes and each node contains the actual values
29  * stored by the user. As such the iterator needs to return that value and not the
30  * node as that is part of the private implementation.
31  ****************************************************************************************************************/
32 // \cond DO_NOT_DOCUMENT
33 template <class Node, class T>
34 struct ListIterator {
35  typedef T value_type;
36  typedef T& reference;
37  typedef T* pointer;
38  typedef bidirectional_iterator_tag iterator_category;
39  typedef std::ptrdiff_t difference_type;
40 
41  Node * node;
42  ListIterator() : node(0) {}
43  ListIterator(Node * node) : node(node) {}
44  ListIterator(const ListIterator & copy) { node = copy.node; }
45  reference operator*() { return node->value; }
46  pointer operator->() { return &node->value; }
47  ListIterator & operator++() { node = node->next; return *this; }
48  ListIterator operator++(int junk) { Node * n = node; node = node->next; return n; }
49  ListIterator & operator--() { node = node->prev; return *this; }
50  ListIterator operator--(int junk) { Node * n = node; node = node->prev; return n; }
51  ListIterator & operator=(const ListIterator & rhs) { node = rhs.node; return *this;}
52  bool operator!=(const ListIterator & rhs) { return node != rhs.node; }
53  bool operator==(const ListIterator & rhs) const { return node == rhs.node; }
54 
55  // Related non-members
56  friend inline difference_type operator-(const ListIterator & lhs, const ListIterator & rhs) { return lhs.node - rhs.node; }
57  friend inline const bool operator<(const ListIterator & lhs, const ListIterator & rhs) { return lhs-rhs < 0; }
58  friend inline const bool operator>(const ListIterator & lhs, const ListIterator & rhs) { return lhs-rhs > 0; }
59  friend inline const bool operator<=(const ListIterator & lhs, const ListIterator & rhs) { return lhs-rhs <= 0; }
60  friend inline const bool operator>=(const ListIterator & lhs, const ListIterator & rhs) { return lhs-rhs >= 0; }
61 };
