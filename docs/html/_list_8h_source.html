26 /****************************************************************************************************************
27  * Special iterator for the List container (implemented as a bidirectional iterator.
28  * A list contains a linked list of nodes and each node contains the actual values
29  * stored by the user. As such the iterator needs to return that value and not the
30  * node as that is part of the private implementation.
31  ****************************************************************************************************************/
32 template <class Node, class T>
33 struct ListIterator {
34  typedef T value_type;
35  typedef T& reference;
36  typedef T* pointer;
37  typedef bidirectional_iterator_tag iterator_category;
38  typedef std::ptrdiff_t difference_type;
39 
40  Node * node;
41  ListIterator() : node(0) {}
42  ListIterator(Node * node) : node(node) {}
43  ListIterator(const ListIterator & copy) { node = copy.node; }
44  reference operator*() { return node->value; }
45  pointer operator->() { return &node->value; }
46  ListIterator & operator++() { node = node->next; return *this; }
47  ListIterator operator++(int junk) { Node * n = node; node = node->next; return n; }
48  ListIterator & operator--() { node = node->prev; return *this; }
49  ListIterator operator--(int junk) { Node * n = node; node = node->prev; return n; }
50  ListIterator & operator=(const ListIterator & rhs) { node = rhs.node; return *this;}
51  bool operator!=(const ListIterator & rhs) { return node != rhs.node; }
52  bool operator==(const ListIterator & rhs) const { return node == rhs.node; }
53 
54  // Related non-members
55  friend inline difference_type operator-(const ListIterator & lhs, const ListIterator & rhs) { return lhs.node - rhs.node; }
56  friend inline const bool operator<(const ListIterator & lhs, const ListIterator & rhs) { return lhs-rhs < 0; }
57  friend inline const bool operator>(const ListIterator & lhs, const ListIterator & rhs) { return lhs-rhs > 0; }
58  friend inline const bool operator<=(const ListIterator & lhs, const ListIterator & rhs) { return lhs-rhs <= 0; }
59  friend inline const bool operator>=(const ListIterator & lhs, const ListIterator & rhs) { return lhs-rhs >= 0; }
60 };
