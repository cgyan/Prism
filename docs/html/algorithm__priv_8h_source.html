725 template <class RandomAccessIterator>
726 void
727 p_heapify(RandomAccessIterator node, RandomAccessIterator first, RandomAccessIterator last) {
728 
729  int nodeIndex = node-first;
730  int leftIndex = 2 * nodeIndex + 1;
731  int rightIndex = 2 * nodeIndex + 2;
732 
733  RandomAccessIterator leftNode = first + leftIndex;
734  RandomAccessIterator rightNode = first + rightIndex;
735  RandomAccessIterator largestNode = node;
736 
737  if (leftNode < last && *largestNode < *leftNode)
738  largestNode = leftNode;
739 
740  if (rightNode < last && *largestNode < *rightNode)
741  largestNode = rightNode;
742 
743  if (node != largestNode) {
744  swap(*node, *largestNode);
745  p_heapify(largestNode, first, last);
746  }
