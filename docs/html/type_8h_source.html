37 // conditional_type
38 //============================================================================================
39 /*
40  * conditional_type is a struct that has a single member called type.
41  * If condition is true then type's data type is set to Type1's data type.
42  * If condition is false then type's data type is set to Type2's data type.
43  *
44  * prism::conditional_type<true, int, String>::type t; // type of t is int
45  * prism::conditional_type<false, int, String>::type t; // type of t is String
46  */
47 template <bool condition, class T1, class T2>
48 struct conditional_type {
49  typedef T1 type;
50 };
