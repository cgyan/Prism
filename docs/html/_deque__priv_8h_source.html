74  Self&
75  operator+=(const int n)
76  {
77  int elementIndex = n + (current-start);
78  if (elementIndex >= 0 && elementIndex < prism_deque_bucket_size)
79  current += n;
80  else {

468  typename DequeData<T, Alloc>::iterator
469  fillInsert(iterator pos, const int numElements, const_reference value) {
470  int insertIndex = pos-begin;
471 
472  // insert at beginning
473  if (pos.current == begin.current) {
474  reserveElementsAtFront(numElements);
475  prism::uninitialized_fill(begin, begin+numElements, value);
476  return begin;
477  }
478  // inserting at end
479  else if (pos.current == end.current) {
480  int oldSize = size();
481  reserveElementsAtBack(numElements);
482  prism::uninitialized_fill(begin+oldSize, begin+oldSize+numElements, value);
483  return begin+oldSize;
484  }
485  // inserting somewhere between beginning and end
486  else {
487  bool indexLocatedInFirstHalf = pos-begin < size() / 2;
488 
489  // insertion index is located in first half
490  if (indexLocatedInFirstHalf) {
491  reserveElementsAtFront(numElements);
492  prism::copy_n(begin+numElements, insertIndex, begin);
493  prism::fill(begin+insertIndex, begin+insertIndex+numElements, value);
494  }
495  // insertion index is located in second half
496  else {
497  reserveElementsAtBack(numElements);
498  prism::copy_backward(begin+insertIndex, end-numElements, end);
499  prism::fill(begin+insertIndex, begin+insertIndex+numElements, value);
500  }
501  }
502  return begin+insertIndex;
503  }
